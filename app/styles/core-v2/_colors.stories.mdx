export const buildColors = (colors) => {
  return colors.reduce((acc, v) => {
    const humanized = v
      .split('-')
      .slice(1, v.split('-').length)
      .map((x) => {
        return x.toUpperCase();
      })
      .join(' ');
    acc[humanized] = getComputedStyle(document.documentElement).getPropertyValue('--' + v);
    return acc;
  }, {});
};

# Colors

New color palette use CSS variables.

Note: The most important variables can be used as html classes, calling them as `background-color-[color]-[weight]`.

E.G. `<div class="background-color-primary-50 font-color-gray-900">`

## Neutral Colors

Neutral colors apply to most backgrounds, borders, and shapes in our experiences. They don’t typically have a meaning associated with them, though they can imply things like disabled states or a delimited group.
